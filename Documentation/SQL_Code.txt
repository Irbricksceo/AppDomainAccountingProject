CREATE TABLE IF NOT EXISTS `accounts` (
    `ID` int(11) NOT NULL AUTO_INCREMENT UNIQUE,
    `username` varchar(50) NOT NULL UNIQUE,
    `password` varchar(255) NOT NULL,
    `Fname` varchar(50) NOT NULL,
    `Lname` varchar(50) NOT NULL,
    `StreetAddress` varchar(50) NOT NULL,
    `City` varchar(50) NOT NULL,
    `State`  varchar(50) NOT NULL,
    `Zip` int(5) NOT NULL,
    `DOB` date NOT NULL,
    `SecurityQ1` varchar(255) NOT NULL,
    `SecurityA1` varchar(50) NOT NULL,
    `SecurityQ2` varchar(255) NOT NULL,
    `SecurityA2` varchar(50) NOT NULL,
    `picture` blob,
    `DateCreated` datetime default current_timestamp NOT NULL,
    `userrole`int(1) NOT NULL,
    `PasswordExpire` datetime NOT NULL,
    `Active` int(1) NOT NULL,
    `SuspendStart` datetime NOT NULL,
    `SuspendEnd` datetime NOT NULL,
    `Email` varchar(50) NOT NULL UNIQUE,
    PRIMARY KEY (`ID`)
) ENGINE=InnoDB AUTO_INCREMENT=2 DEFAULT CHARSET=utf8;
